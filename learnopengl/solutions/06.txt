// Vertex shader
#version 330 core
layout (location = 0) in vec3 aPos;
layout (location = 1) in vec3 aColor;

out vec3 ourColor;
out vec3 ourPos;

void main()
{
	gl_Position = vec4(aPos, 1.0f);
	ourColor = aColor;
	ourPos = aPos;
}

// Fragment shader
#version 330 core
out vec4 FragColor;
in vec3 ourPos;

void main()
{
	FragColor = vec4(ourPos, 1.0f);
}
